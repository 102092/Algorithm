## 우선순위 큐

```java
package app;

import java.io.*;
import java.io.IOException;
import java.util.PriorityQueue;


public class Main {
    
    
    public static void main(String[] args) throws IOException {        
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringBuilder sb = new StringBuilder();

        PriorityQueue<Integer> pq = new PriorityQueue<>();
        int n = Integer.parseInt(br.readLine());

        for (int i = 0; i <n; i++) {
            pq.add(Integer.parseInt(br.readLine()));
        }

        while (!pq.isEmpty()) {
            if(pq.size() >1){
                sb.append(pq.poll()+"\n");
            }
            else{
                sb.append(pq.poll());
            }
            
            
        }
        System.out.println(sb);


    }
}
```



## 합병정렬

```java
package app;

import java.io.*;
import java.io.IOException;


public class Main {
    static int result[];
    
    public static void main(String[] args) throws IOException {        
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringBuilder sb = new StringBuilder();

        int n = Integer.parseInt(br.readLine().trim());

        int[] arr = new int[n];
        result = new int[n];
        for (int i = 0; i < n; i++) {
            arr[i] = Integer.parseInt(br.readLine().trim());
        }

        merge_sort(arr, 0, n-1);

        for (int i = 0; i < n; i++) {
            if (i < n-1) {
                sb.append(result[i]+"\n");    
            } else {
                sb.append(result[i]);
            }        
            
        }
        System.out.println(sb);
        


    }

    public static void merge(int[] input, int left, int mid, int right){
        int a, b, c;
        a = left;
        b = mid +1;
        c = left;

        while (a<=mid || b<=right ) {
            if(a<=mid && b<=right){
                if (input[a] <= input[b]) {
                    result[c] = input[a++];
                }
                else{
                    result[c] = input[b++];
                }
            }
            else if(a <= mid && b > right){
                result[c] = input[a++];
            }
            else{
                result[c] = input[b++];
            }
            c++;
        }
        for(int i=left; i<right+1; i++){
            input[i] = result[i];
        }
    }

    public static void merge_sort(int[] input, int left, int right){
        int mid;
        if(left < right){
            mid = (left+right)/2;
            merge_sort(input, left, mid);
            merge_sort(input, mid+1, right);
            merge(input, left, mid, right);
        }
    }
}
```





## 퀵 정렬

```java
package app;

import java.io.*;
import java.io.IOException;

public class Main {
    
    
    public static void main(String[] args) throws IOException {        
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringBuilder sb = new StringBuilder();

        int n = Integer.parseInt(br.readLine().trim());

        int[] result = new int[n];        
        for (int i = 0; i < n; i++) {
            result[i] = Integer.parseInt(br.readLine().trim());
        }

        QuickSort(result, 0, n-1);

        for (int i = 0; i < n; i++) {
            if (i < n-1) {
                sb.append(result[i]+"\n");    
            } else {
                sb.append(result[i]);
            }        
            
        }
        System.out.println(sb);
        


    }

    public static void QuickSort(int[] input, int left, int right) {
        if(left > right) return;
        int pivot = input[right]; //맨끝
        int a = left;
        int b = right-1;

        while(a <=b){
            while (a<=b && input[a] <=pivot) { 
                a++;                
            }
            while (a<=b && input[b] >=pivot) { 
                b--;                
            }
            if(a<b) SWAP(input, a,b);
        }
        SWAP(input, a, right);
        QuickSort(input, left, a-1);
        QuickSort(input, a+1, right);

        
    }

    public static void SWAP(int[] input, int a, int b) {
        int temp = input[a];
        input[a] = input[b];
        input[b] = temp;
    }

}
```
